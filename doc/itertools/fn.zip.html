Converts the arguments to iterators and zips them.

IntoIterator enabled version of Iterator::zip.

Â§Example

use itertools::zip;

let mut result: Vec<(i32, char)> = Vec::new();

for (a, b) in zip(&[1, 2, 3, 4, 5], &['a', 'b', 'c']) {
    result.push((*a, *b));
}
assert_eq!(result, vec![(1, 'a'),(2, 'b'),(3, 'c')]);
